Ufunguo umekubaliwa! Msimbo wako wa ESP32 uko hapa chini.
Nakili msimbo huu uweke kwenye Arduino IDE, badilisha WiFi na Device ID yako, kisha pakia kwenye ESP32. 
#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <DHT.h>

// ══ BADILISHA HAPA ══
const char* ssid      = "WIFI_YAKO";
const char* password  = "NENOSIRI_WIFI";
const char* DEVICE_ID = "ESP32-XXXXXX";  // Device ID uliyosajili kwenye app
const char* USER_ID   = "WEKA_USER_ID";  // Nakili kutoka kwa app → Vifaa → User ID

// ══ SUPABASE CONFIG ══
const char* SUPABASE_URL = "https://klhrdnpvkddftmfzmwxt.supabase.co";
const char* SUPABASE_KEY = "sb_publishable_lCJvWV1b_tEBFyCy1qV5FA_mnambYwM";

// ══ PIN ZA SENSOR ══
#define DHTPIN       4     // DHT22 → GPIO4
#define DHTTYPE      DHT22
#define SOIL_PIN     34    // Soil moisture → GPIO34 (analog)
#define PH_PIN       35    // pH sensor → GPIO35 (analog)
#define EC_PIN       32    // EC sensor → GPIO32 (analog)
#define RELAY_PIN    26    // Relay/pump → GPIO26
#define LED_PIN      2     // Built-in LED

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(115200);
  dht.begin();
  pinMode(RELAY_PIN, OUTPUT);
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, LOW);

  WiFi.begin(ssid, password);
  Serial.print("Inaunganisha WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500); Serial.print(".");
    digitalWrite(LED_PIN, !digitalRead(LED_PIN));
  }
  digitalWrite(LED_PIN, HIGH);
  Serial.println("\nWiFi imeunganishwa!");
  Serial.print("Device ID: "); Serial.println(DEVICE_ID);
  Serial.print("IP Address: "); Serial.println(WiFi.localIP());
}

void loop() {
  if (WiFi.status() == WL_CONNECTED) {

    // ── Soma sensor ──
    float temperature = dht.readTemperature();
    float humidity    = dht.readHumidity();
    if (isnan(temperature)) temperature = 25.0;
    if (isnan(humidity))    humidity    = 65.0;

    // Soil moisture (0-4095 raw → 0-100%)
    int   rawMoist  = analogRead(SOIL_PIN);
    float moisture  = map(rawMoist, 4095, 1500, 0, 100);
    moisture        = constrain(moisture, 0, 100);

    // pH (calibrate kwa sensor yako)
    int   rawPH = analogRead(PH_PIN);
    float ph    = (rawPH / 4095.0) * 14.0;
    ph          = constrain(ph, 0, 14);

    // EC (calibrate kwa sensor yako)
    int   rawEC = analogRead(EC_PIN);
    float ec    = (rawEC / 4095.0) * 5.0;

    // Pump status
    bool pumpOn = digitalRead(RELAY_PIN) == HIGH;

    Serial.printf("Temp:%.1f Hum:%.1f Moist:%.1f pH:%.2f EC:%.2f Pump:%d\n",
                  temperature, humidity, moisture, ph, ec, pumpOn);

    // ── Tuma kwa Supabase ──
    HTTPClient http;
    String url = String(SUPABASE_URL) + "/rest/v1/sensor_readings";
    http.begin(url);
    http.addHeader("Content-Type",  "application/json");
    http.addHeader("apikey",        SUPABASE_KEY);
    http.addHeader("Authorization", String("Bearer ") + SUPABASE_KEY);
    http.addHeader("Prefer",        "return=minimal");

    StaticJsonDocument<300> doc;
    doc["user_id"]     = USER_ID;
    doc["device_id"]   = DEVICE_ID;
    doc["temperature"] = round(temperature * 10) / 10.0;
    doc["moisture"]    = round(moisture * 10) / 10.0;
    doc["humidity"]    = round(humidity * 10) / 10.0;
    doc["ph"]          = round(ph * 100) / 100.0;
    doc["ec"]          = round(ec * 100) / 100.0;
    doc["pump_status"] = pumpOn;

    String body;
    serializeJson(doc, body);
    int httpCode = http.POST(body);
    Serial.print("Supabase HTTP: "); Serial.println(httpCode);
    http.end();

    // ── Auto-pump: washa pampu ukijaa chini ya 30% ──
    if (moisture < 30.0) {
      digitalWrite(RELAY_PIN, HIGH);
      Serial.println("Pampu: WASHA (unyevu chini)");
    } else if (moisture > 70.0) {
      digitalWrite(RELAY_PIN, LOW);
      Serial.println("Pampu: ZIMA (unyevu wa kutosha)");
    }
  }

  delay(10000); // Tuma kila sekunde 10
}
